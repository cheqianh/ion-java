        byte [] _buffer;

            _pos = 0;
            _len = buffer.length;

        @Override
        public final int getByte(int pos) {

        public OffsetBufferStream(byte[] buffer, int start, int max)

            _buffer = buffer;
            _pos = start;
            _start = start;
            _end = start + max;
